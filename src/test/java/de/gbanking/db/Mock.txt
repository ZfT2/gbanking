	public boolean addNewBankAccessMock(BankAccess bankAccess) {

		log.info("addNewBankAccessMock called.");

		List<BankAccount> bankAccountList = new ArrayList<>();
		try {
			
			bankAccess.setBankName("Bank12");
			bankAccess.setHbciURL("https://Bank12.de/hbci");
			bankAccess.setBpdVersion("0.1");
			bankAccess.setUpdVersion("0.2");
			bankAccess.setUserId("boener02");
			bankAccess.setUpdatedAt(Calendar.getInstance());

			Konto[] konten = getAccountsMock();
			if (konten == null || konten.length == 0) {
				log.error("Keine Konten ermittelbar");
			} else {
				log.info("Anzahl Konten: {}", konten.length);
				log.info(messages.getFormattedMessage("LOG_INFO_ACCOUNT_COUNT", konten.length));
			}
			
			for (Konto konto : konten) {

				log.info(
						"Konto: acctype {}, bic {}, blz {}, country {}, creditorid {}, curr {}, customerid {}, iban {}, limit {}, name {}, name2 {}, number {}, subnumber {}, type {}, isSEPAAccount? {}",
						konto.acctype, konto.bic, konto.blz, konto.country, konto.creditorid, konto.curr,
						konto.customerid, konto.iban, konto.limit, konto.name, konto.name2, konto.number,
						konto.subnumber, konto.type, konto.isSEPAAccount());

				log.info("Konto: GVs:");
				List<?> allowedGVs = konto.allowedGVs;
				for (Object gv : allowedGVs) {
					log.info("GV: {}", gv);
				}
				bankAccountList.add(HbciMapper.mapKontoToBankAccount("Bank12", konto));
			}

			Properties upd = new Properties();
			upd.put("updK01", "VU01");
			upd.put("updK02", "VU02");
			
			Properties bpd = new Properties();
			bpd.put("bpdK01", "VB01");
			bpd.put("bpdK02", "VB02");
			
			bankAccess.setUpd(upd);
			bankAccess.setBpd(bpd);
			bankAccess.setAccounts(bankAccountList);
			
			HBCIExecStatus status  = new HBCIExecStatus();
			
			if (!status.isOK())
				log.error("HBCI Error, Status: {}", status);
			
			return true;


		} finally {
			;
		}
		
	}
